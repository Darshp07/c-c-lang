#include <stdio.h>

int main() {
    int n;
    int sum ;

    printf("Enter your number: ");
    scanf("%d", &n);

    for (int i = 1; i <= n; i++) {
        int i = n % 10;
        sum += i;
      n /= 10;
    }

    printf("The summation  numbers is  %d.\n", sum);

    return 0;
}
